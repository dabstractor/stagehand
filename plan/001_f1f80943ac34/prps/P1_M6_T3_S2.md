## Why

- **Defensibility proof (PRD §18.1, §20.1 layer 3)**: this is the suite that proves the real `CommitStaged` creates a real commit with the correct TREE/PARENT/message on success and leaves the repo byte-for-byte unchanged (HEAD + index) on every failure path. The unit tests (generate_test.go) prove the two-nested-loop LOGIC via stubs; this suite proves the WIRING against real git + a real exec'd subprocess.
- **Feeds P1.M6.T3.S3**: the §18.1 invariant tests (idempotent index, atomic HEAD, snapshot immutability) compose these same failure-path scenarios. This suite establishes the realistic per-path scenarios + assertion mechanics the invariant suite generalizes.
- **Integration point**: consumed by no production code (test-only artifact). Its OUTPUT is the proof that M2+M3+M4+M5 integrate correctly inside M6's `CommitStaged`.

package generate
import (

func gitRun(t *testing.T, dir string, args ...string) string {
	t.Helper()
	cmd := exec.Command("git", args...)
	cmd.Dir = dir
	var stdout, stderr bytes.Buffer
	cmd.Stdout = &stdout
	cmd.Stderr = &stderr
	if err := cmd.Run(); err != nil {
		t.Fatalf("git %s: %v\nstderr: %s", strings.Join(args, " "), err, stderr.String())
	}
	return stdout.String()
}
func newTempRepo(t *testing.T) string {
	t.Helper()
	dir := t.TempDir()
	gitRun(t, dir, "init", "-q")
	gitRun(t, dir, "config", "user.email", "stagehand@example.com")
	gitRun(t, dir, "config", "user.name", "Stagehand Test")
	gitRun(t, dir, "config", "commit.gpgsign", "false")
	return dir
}
func writeStage(t *testing.T, dir, path, content string) {
	t.Helper()
}
func seedCommit(t *testing.T, dir, msg string) {
	t.Helper()
}
type headMoverRunner struct {
	inner runner
	dir   string
	moved bool
}
func (r *headMoverRunner) Run(ctx context.Context, m provider.Manifest, model, prov, sys, payload string) (string, error) {
	if !r.moved {
		r.moved = true
		tree := strings.TrimSpace(gitRunTest(r.dir, "rev-parse", "HEAD^{tree}"))
		parent := strings.TrimSpace(gitRunTest(r.dir, "rev-parse", "HEAD"))
		newc := strings.TrimSpace(gitRunTest(r.dir, "commit-tree", "-p", parent, "-m", "concurrent commit elsewhere", tree))
		gitRunTest(r.dir, "update-ref", "HEAD", newc)
	}
	return r.inner.Run(ctx, m, model, prov, sys, payload)
}
func (r *headMoverRunner) Parse(raw string, m provider.Manifest) (string, bool) { return r.inner.Parse(raw, m) }
func e2eDeps(t *testing.T, dir string, script []StubResponse, cfg config.Config, stdout, stderr *bytes.Buffer) Deps {
	t.Helper()
	manifest := NewStubManifest(t, StubConfig{
		Script:    script,
		StateFile: filepath.Join(t.TempDir(), "c"), 
	})
	g, err := git.New(dir)
	if err != nil {
		t.Fatalf("git.New: %v", err)
	}
	return Deps{
		Git: g, Runner: provider.NewExecutor(""), Manifest: manifest,
		Config: cfg, Output: ui.NewOutput(stdout, stderr, false, true),
	}
}
